
    public class Employee {
        [Required]
        public string FirstName { set; get; }
        [Required]
        public string LastName { set; get; }

        public bool IsValid() {
            int error = 0;
            if (String.IsNullOrEmpty(this.FirstName))
                error += 1;
            if (String.IsNullOrEmpty(this.LastName))
                error += 1;

            return (error == 0);
        }
    }
